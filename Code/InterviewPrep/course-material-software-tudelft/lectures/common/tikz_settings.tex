\newcommand\ifMaxFlow[4]{
	\begingroup
		\pgfmathsetmacro{\flow}{#1}
		\pgfmathsetmacro{\cap}{#2}
		\pgfmathparse{ifthenelse(\flow == \cap,1,0)}
		\ifdim\pgfmathresult pt= 1 pt
			 #3
			\else
			 #4
		\fi
	\endgroup
}

\newcommand\ifSomeFlow[3]{
	\begingroup
		\pgfmathsetmacro{\flow}{#1}
		\pgfmathparse{ifthenelse(\flow > 0,1,0)}
		\ifdim\pgfmathresult pt= 1 pt
			 #2
			\else
			 #3
		\fi
	\endgroup
}
